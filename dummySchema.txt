CREATE DATABASE academic_system;
USE academic_system;

-- 1. USERS TABLE
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    emp_id VARCHAR(20) UNIQUE,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    role ENUM('admin', 'dean', 'HOD', 'staff') NOT NULL,
    dept VARCHAR(100)
);

-- 2. COURSE TABLE
CREATE TABLE course (
    id INT AUTO_INCREMENT PRIMARY KEY,
    course_code VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(150) NOT NULL,
    credit INT CHECK (credit > 0),
    syllabus_link VARCHAR(255),
    faculty INT,
    year YEAR,
    semester ENUM('1','2','3','4','5','6','7','8'),
    FOREIGN KEY (faculty) REFERENCES users(id)
);

-- 3. STUDENT TABLE
CREATE TABLE student (
    rollno VARCHAR(20) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    dept VARCHAR(100),
    programme VARCHAR(100)
);

-- 4. ENROLMENT TABLE
CREATE TABLE enrolment (
    id INT AUTO_INCREMENT PRIMARY KEY,
    course_id INT,
    rollno VARCHAR(20),
    FOREIGN KEY (course_id) REFERENCES course(id) ON DELETE CASCADE,
    FOREIGN KEY (rollno) REFERENCES student(rollno) ON DELETE CASCADE
);

-- 5. ASSESSMENT TABLE
CREATE TABLE assessment (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20),
    course_id INT,
    marks DECIMAL(5,2),
    grade CHAR(2),
    FOREIGN KEY (student_id) REFERENCES student(rollno),
    FOREIGN KEY (course_id) REFERENCES course(id)
);

-- 6. TEST TABLE
CREATE TABLE test (
    id INT AUTO_INCREMENT PRIMARY KEY,
    course_id INT,
    name VARCHAR(100),
    full_marks INT CHECK (full_marks > 0),
    pass_marks INT CHECK (pass_marks >= 0),
    question_link VARCHAR(255),
    is_verified BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (course_id) REFERENCES course(id)
);

-- 7. QUESTION TABLE
CREATE TABLE question (
    id INT AUTO_INCREMENT PRIMARY KEY,
    test_id INT,
    map ENUM('C1','C2','C3','C4','C5','C6'),
    max_marks INT CHECK (max_marks > 0),
    FOREIGN KEY (test_id) REFERENCES test(id)
);

-- 8. RAW TABLE (for raw marks)
CREATE TABLE raw (
    id INT AUTO_INCREMENT PRIMARY KEY,
    test_id INT,
    student_id VARCHAR(20),
    question_id INT,
    marks DECIMAL(5,2),
    FOREIGN KEY (test_id) REFERENCES test(id),
    FOREIGN KEY (student_id) REFERENCES student(rollno),
    FOREIGN KEY (question_id) REFERENCES question(id)
);

-- 9. COMPILED_CO TABLE
CREATE TABLE compiled_co (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20),
    test_id INT,
    CO1 DECIMAL(5,2),
    CO2 DECIMAL(5,2),
    CO3 DECIMAL(5,2),
    CO4 DECIMAL(5,2),
    CO5 DECIMAL(5,2),
    CO6 DECIMAL(5,2),
    FOREIGN KEY (student_id) REFERENCES student(rollno),
    FOREIGN KEY (test_id) REFERENCES test(id)
);
